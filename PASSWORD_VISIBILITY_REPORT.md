#### **🔧 JavaScript 功能**
```javascript
// 切換密碼顯示/隱藏
function togglePassword(inputId, button) {
    const input = document.getElementById(inputId);
    const icon = document.getElementById(inputId + '-icon');
    
    if (input.type === 'password') {
        input.type = 'text';           // 切換為明文
        icon.classList.remove('fa-eye');   // 移除眼睛圖標
        icon.classList.add('fa-eye-slash'); // 添加眼睛斜線圖標
    } else {
        input.type = 'password';         // 切換為密碼
        icon.classList.remove('fa-eye-slash'); // 移除眼睛斜線圖標
        icon.classList.add('fa-eye');      // 添加眼睛圖標
    }
}
```
